        var Count_Occ = (s) => 
        {
            let newString = '';
            s = s.toLowerCase().split('').sort().join('');
            for(let i = 0;i<(s.length - 1);i++) {
            let count = 0;
            for(let j =i+1;j< s.length;j++) {
                if(s[i] === s[j]) {
                count++;
                }
            }
            if(count>0 && !newString.includes(s[i])) {
                newString+=s[i];
                newString+=count+1;
            }
            }
            
            return newString;
        };

7.You are given a string S consisting of two letters A and D,where each character represent the winner of N games played between Aditya and Danish, where letter A represents the win of Aditya and letter D represents the win of Danish.You need to find out the that which player wins the maximum number of games or there is a draw between them

        var Game_Winner = (S) => 
        {
            let countOfAditya = 0;
            let countOfDanish = 0;
            for(let i = 0; i < S.length; i++) {
                if(S[i] === 'A') {
                    countOfAditya++;
                } else {
                    countOfDanish++;
                }
            }
            
            if(countOfAditya === countOfDanish) {
                return 'Draw'
            } else if(countOfAditya > countOfDanish) {
                return 'Aditya'
            } else {
                return 'Danish'
            }
        }
